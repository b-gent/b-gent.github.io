412  iterator insert_aux(iterator pos, const_iterator first, const_iterator last) {
413  assert(_storage <= pos && pos <= _storage + _size);
414  assert(first <= last);
415  const size_type n = last - first;
416  if (n) {
417  const size_type idx = pos - _storage;
418  if (_size + n > _capacity || (_storage <= first && first <= _storage + _size)) {
419  T *const oldStorage = _storage;
420 
421  // If there is not enough space, allocate more.
422  // Likewise, if this is a self-insert, we allocate new
423  // storage to avoid conflicts.
424  allocCapacity(roundUpCapacity(_size + n));
425 
426  // Copy the data from the old storage till the position where
427  // we insert new data
428  uninitialized_copy(oldStorage, oldStorage + idx, _storage);
429  // Copy the data we insert
430  uninitialized_copy(first, last, _storage + idx);
431  // Afterwards, copy the old data from the position where we
432  // insert.
433  uninitialized_copy(oldStorage + idx, oldStorage + _size, _storage + idx + n);
434 
435  freeStorage(oldStorage, _size);
436  } else if (idx + n <= _size) {
437  // Make room for the new elements by shifting back
438  // existing ones.
439  // 1. Move a part of the data to the uninitialized area
440  uninitialized_copy(_storage + _size - n, _storage + _size, _storage + _size);
441  // 2. Move a part of the data to the initialized area
442  copy_backward(pos, _storage + _size - n, _storage + _size);
443 
444  // Insert the new elements.
445  copy(first, last, pos);
446  } else {
447  // Copy the old data from the position till the end to the new
448  // place.
449  uninitialized_copy(pos, _storage + _size, _storage + idx + n);
450 
451  // Copy a part of the new data to the position inside the
452  // initialized space.
453  copy(first, first + (_size - idx), pos);
454 
455  // Copy a part of the new data to the position inside the
456  // uninitialized space.
457  uninitialized_copy(first + (_size - idx), last, _storage + _size);
458  }
459 
460  // Finally, update the internal state
461  _size += n;
462  }
463  return pos;
464  }

515  T *bsearchMin(CompareArgType key) {
516  uint start_ = 0, end_ = this->_size;
517  int result;
518 
519  while (start_ < end_) {
520  uint mid = start_ + (end_ - start_) / 2;
521 
522  result = this->_comparator(key, this->_storage[mid]);
523  if (result < 0)
524  end_ = mid;
525  else
526  start_ = mid + 1;
527  }
528 
529  return &this->_storage[start_];
530  }
