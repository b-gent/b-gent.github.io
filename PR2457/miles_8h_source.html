288  struct MidiChannelEntry {
289  // True if this channel is locked. A locked channel will
290  // only accept MIDI messages from the source that locked it.
291  bool locked;
292  // The channel in the MIDI data of the lock source that
293  // is assigned to this locked output channel. This is a
294  // reverse lookup for MidiSource::channelMap.
295  // -1 if the channel is not locked.
296  int8 lockDataChannel;
297  // True if this channel is protected from locking.
298  // The channel can still be locked, but unprotected
299  // channels will be prioritized.
300  bool lockProtected;
301  // The source that protected this channel from locking.
302  // -1 if the channel is not protected.
303  int8 protectedSource;
304 
305  // The number of notes currently active on the channel.
306  uint8 activeNotes;
307 
308  // The MIDI controller values currently used by the channel.
309  MidiChannelControlData currentData;
310  // The MIDI controller values set by the sources which are
311  // not currently using the channel because it is locked.
312  // These values will be set on the channel when the channel
313  // is unlocked.
314  MidiChannelControlData unlockData;
315 
316  MidiChannelEntry() : locked(false),
317  lockDataChannel(-1),
