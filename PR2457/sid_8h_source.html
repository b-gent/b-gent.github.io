36 // We could have used the smallest possible data type for each SID register,
37 // however this would give a slower engine because of data type conversions.
38 // An int is assumed to be at least 32 bits (necessary in the types reg24,
39 // cycle_count, and sound_sample). GNU does not support 16-bit machines
40 // (GNU Coding Standards: Portability between CPUs), so this should be
41 // a valid assumption.
42

45 typedef unsigned int reg12;
46 typedef unsigned int reg16;
47 typedef unsigned int reg24;

51 typedef sound_sample fc_point[2];
