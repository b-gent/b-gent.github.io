51 template<typename T> inline T CLIP(T v, T amin, T amax)
52  {
53 #if !defined(RELEASE_BUILD)
54  // debug builds use this assert to pinpoint
55  // any problematic cases, where amin and amax
56  // are incorrectly ordered
57  // and thus CLIP() would return an invalid result
58  assert(amin <= amax);
59 #endif
60  if (v < amin) return amin;
61  else if (v > amax) return amax;
62  return v;
63  }
