69  void beginDialog() {
70  // If we were in fullscreen mode, switch back
71  _wasFullscreen = g_system->getFeatureState(OSystem::kFeatureFullscreenMode);
72  if (_wasFullscreen) {
73  g_system->beginGFXTransaction();
74  g_system->setFeatureState(OSystem::kFeatureFullscreenMode, false);
75  g_system->endGFXTransaction();
76  }
77  }
78 
82  void endDialog() {
83  // While the native file browser is open, any input events (e.g. keypresses) are
84  // still received by the application. With SDL backend for example this results in the
85  // events beeing queued and processed after we return, thus dispatching events that were
86  // intended for the native file browser. For example: pressing Esc to cancel the native
87  // file browser would cause the application to quit in addition to closing the
88  // file browser. To avoid this happening clear all pending events.
89  g_system->getEventManager()->getEventDispatcher()->clearEvents();
90 
91  // If we were in fullscreen mode, switch back
92  if (_wasFullscreen) {
93  g_system->beginGFXTransaction();
94  g_system->setFeatureState(OSystem::kFeatureFullscreenMode, true);
95  g_system->endGFXTransaction();
96  }
97  }
98 };
