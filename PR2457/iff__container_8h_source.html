175  IFFChunkNav() : _input(nullptr) {

177  void setInputStream(ReadStream *input) {
178  _input = input;
179  size = _bytesRead = 0;
180  }
181  void incBytesRead(uint32 inc) {
182  _bytesRead += inc;
183  if (_bytesRead > size) {
184  error("Chunk overread");
185  }
186  }
187  void readHeader() {
188  id = _input->readUint32BE();
189  size = _input->readUint32BE();
190  _bytesRead = 0;
191  }
192  bool hasReadAll() const {
193  return (size - _bytesRead) == 0;
194  }
195  void feed() {
196  if (size % 2) {
197  size++;
198  }
199  while (!hasReadAll()) {
200  readByte();
201  }
202  }
203  // ReadStream implementation
204  bool eos() const { return _input->eos(); }
205  bool err() const { return _input->err(); }
206  void clearErr() { _input->clearErr(); }
207 
208  uint32 read(void *dataPtr, uint32 dataSize) {
209  incBytesRead(dataSize);
210  return _input->read(dataPtr, dataSize);
211  }
212  };
