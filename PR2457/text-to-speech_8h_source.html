177  bool say(const String &str, Action action, String charset = "UTF-8") {
178  Encoding speakWithCustomCharset("UTF-32", charset);
179  char *res = speakWithCustomCharset.convert(str.c_str(), str.size());
180  U32String textToSpeak(reinterpret_cast<uint32*>(res));
181  free(res);
182 
183  return say(textToSpeak, action);
184  }
