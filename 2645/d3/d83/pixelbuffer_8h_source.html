133  inline void setPixelAt(int pixel, uint32 value) {
134  switch (_format.bytesPerPixel) {
135  case 2:
136  ((uint16 *) _buffer)[pixel] = value;
137  return;
138  case 3:
139  pixel *= 3;
140 #if defined(SCUMM_BIG_ENDIAN)
141  _buffer[pixel + 0] = (value >> 16) & 0xFF;
142  _buffer[pixel + 1] = (value >> 8) & 0xFF;
143  _buffer[pixel + 2] = value & 0xFF;
144 #elif defined(SCUMM_LITTLE_ENDIAN)
145  _buffer[pixel + 0] = value & 0xFF;
146  _buffer[pixel + 1] = (value >> 8) & 0xFF;
147  _buffer[pixel + 2] = (value >> 16) & 0xFF;
148 #endif
149  return;
150  case 4:
151  ((uint32 *) _buffer)[pixel] = value;
152  return;
153  }
154  error("setPixelAt: Unhandled bytesPerPixel %i", _format.bytesPerPixel);
155  }
