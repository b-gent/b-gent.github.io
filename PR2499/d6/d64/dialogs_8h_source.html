78  void beginDialog() {
79  // If we were in fullscreen mode, switch back
80  _wasFullscreen = g_system->getFeatureState(OSystem::kFeatureFullscreenMode);
81  if (_wasFullscreen) {
82  g_system->beginGFXTransaction();
83  g_system->setFeatureState(OSystem::kFeatureFullscreenMode, false);
84  g_system->endGFXTransaction();
85  }
86  }

91  void endDialog() {
92  // While the native file browser is open, any input events (e.g. keypresses) are
93  // still received by the application. With SDL backend for example this results in the
94  // events beeing queued and processed after we return, thus dispatching events that were
95  // intended for the native file browser. For example: pressing Esc to cancel the native
96  // file browser would cause the application to quit in addition to closing the
97  // file browser. To avoid this happening clear all pending events.
98  g_system->getEventManager()->getEventDispatcher()->clearEvents();
99 
100  // If we were in fullscreen mode, switch back
101  if (_wasFullscreen) {
102  g_system->beginGFXTransaction();
103  g_system->setFeatureState(OSystem::kFeatureFullscreenMode, true);
104  g_system->endGFXTransaction();
105  }
106  }
107 };
