183  struct MidiChannelEntry {
184  // True if this channel is locked. A locked channel will
185  // only accept MIDI messages from the source that locked it.
186  bool locked;
187  // The channel in the MIDI data of the lock source that
188  // is assigned to this locked output channel. This is a
189  // reverse lookup for MidiSource::channelMap.
190  // -1 if the channel is not locked.
191  int8 lockDataChannel;
192  // True if this channel is protected from locking.
193  // The channel can still be locked, but unprotected
194  // channels will be prioritized.
195  bool lockProtected;
196  // The source that protected this channel from locking.
197  // -1 if the channel is not protected.
198  int8 protectedSource;
199 
200  // The number of notes currently active on the channel.
201  uint8 activeNotes;
202 
203  // The MIDI controller values currently used by the channel.
204  MilesMidiChannelControlData *currentData;
205  // The MIDI controller values set by the sources which are
206  // not currently using the channel because it is locked.
207  // These values will be set on the channel when the channel
208  // is unlocked.
209  MilesMidiChannelControlData *unlockData;
210 
211  MidiChannelEntry() : locked(false),
212  lockDataChannel(-1),
213  lockProtected(false),
214  protectedSource(-1),
215  activeNotes(0),
216  currentData(0),
217  unlockData(0) { }
