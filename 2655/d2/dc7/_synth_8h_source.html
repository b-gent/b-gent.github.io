240  static inline Bit16s clipSampleEx(Bit32s sampleEx) {
241  // Clamp values above 32767 to 32767, and values below -32768 to -32768
242  // FIXME: Do we really need this stuff? I think these branches are very well predicted. Instead, this introduces a chain.
243  // The version below is actually a bit faster on my system...
244  //return ((sampleEx + 0x8000) & ~0xFFFF) ? Bit16s((sampleEx >> 31) ^ 0x7FFF) : (Bit16s)sampleEx;
245  return ((-0x8000 <= sampleEx) && (sampleEx <= 0x7FFF)) ? Bit16s(sampleEx) : Bit16s((sampleEx >> 31) ^ 0x7FFF);
246  }

317  // Configures the SysEx storage of the internal MIDI event queue.
318  // Supplying 0 in the storageBufferSize argument makes the SysEx data stored
319  // in multiple dynamically allocated buffers per MIDI event. These buffers are only disposed
320  // when a new MIDI event replaces the SysEx event in the queue, thus never on the rendering thread.
321  // This is the default behaviour.
322  // In contrast, when a positive value is specified, SysEx data will be stored in a single preallocated buffer,
323  // which makes this kind of storage safe for use in a realtime thread. Additionally, the space retained
324  // by a SysEx event, that has been processed and thus is no longer necessary, is disposed instantly.
325  // Note, the queue is flushed and recreated in the process so that its size remains intact.
