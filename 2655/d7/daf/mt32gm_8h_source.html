223  struct MidiSource {
224  // Whether this source sends music or SFX MIDI data.
225  SourceType type;
226  // The source volume (relative volume for this source as defined by the game).
227  // Default is the default neutral value (255).
228  uint16 volume;
229  // The source volume level at which no scaling is performed (volume as defined
230  // in MIDI data is used directly). Volume values below this decrease volume,
231  // values above increase volume (up to the maximum MIDI channel volume).
232  // Set this to match the volume values used by the game engine to avoid having
233  // to convert them. Default value is 255; minimum value is 1.
234  uint16 neutralVolume;
235  // The volume level at which the fade started.
236  uint16 fadeStartVolume;
237  // The target volume level for the fade.
238  uint16 fadeEndVolume;
239  // How much us has passed since the start of the fade.
240  int32 fadePassedTime;
241  // The total duration of the fade (us).
242  int32 fadeDuration;
243  // The mapping of MIDI data channels to output channels for this source.
244  int8 channelMap[MIDI_CHANNEL_COUNT];
245  // Bitmask specifying which MIDI channels are available for use by this source.
246  uint16 availableChannels;
