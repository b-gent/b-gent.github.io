158  Common::String getIconPath(const Common::String &target, const Common::String &extension) {
159  // We first try to look for a iconspath configuration variable then
160  // fallback to the extra path
161  //
162  // Icons can be either in a subfolder named "icons" or directly in the path
163 
164  Common::String iconsPath = ConfMan.get("iconspath");
165  Common::String extraPath = ConfMan.get("extrapath");
166 
167  Common::String targetIcon = target + extension;
168  Common::String qualifiedIcon = ConfMan.get("engineid") + "-" + ConfMan.get("gameid") + extension;
169  Common::String gameIcon = ConfMan.get("gameid") + extension;
170 
171 #define TRY_ICON_PATH(path) { \
172 Common::FSNode node((path)); \
173 if (node.exists()) \
174 return (path); \
175 }
176  if (!iconsPath.empty()) {
177  TRY_ICON_PATH(iconsPath + "/" + targetIcon);
178  TRY_ICON_PATH(iconsPath + "/" + qualifiedIcon);
179  TRY_ICON_PATH(iconsPath + "/" + gameIcon);
180  TRY_ICON_PATH(iconsPath + "/icons/" + targetIcon);
181  TRY_ICON_PATH(iconsPath + "/icons/" + qualifiedIcon);
182  TRY_ICON_PATH(iconsPath + "/icons/" + gameIcon);
183  }
184 
185  if (!extraPath.empty()) {
186  TRY_ICON_PATH(extraPath + "/" + targetIcon);
187  TRY_ICON_PATH(extraPath + "/" + qualifiedIcon);
188  TRY_ICON_PATH(extraPath + "/" + gameIcon);
189  TRY_ICON_PATH(extraPath + "/icons/" + targetIcon);
190  TRY_ICON_PATH(extraPath + "/icons/" + qualifiedIcon);
191  TRY_ICON_PATH(extraPath + "/icons/" + gameIcon);
192  }
193 #undef TRY_ICON_PATH
194 
195  return "";
196  }
197 };
