86 // Our 'specialized' 'fill' template for char, signed char and unsigned char arrays.
87 // Since C++ doesn't support partial specialized template functions (currently) we
88 // are going this way...

187 template<typename T, class StrictWeakOrdering>
188 T sortPartition(T first, T last, T pivot, StrictWeakOrdering &comp) {
189  --last;
190  if (pivot != last)
191  SWAP(*pivot, *last);
192 
193  T sorted;
194  for (sorted = first; first != last; ++first) {
195  if (!comp(*last, *first)) {
196  if (first != sorted)
197  SWAP(*first, *sorted);
198  ++sorted;
199  }
200  }
201 
202  if (last != sorted)
203  SWAP(*last, *sorted);
204  return sorted;
205 }

226 template<typename T, class StrictWeakOrdering>
227 void sort(T first, T last, StrictWeakOrdering comp) {
228  if (first == last)
229  return;
230 
231  T pivot = sortChoosePivot(first, last);
232  pivot = sortPartition(first, last, pivot, comp);
233  sort<T, StrictWeakOrdering>(first, pivot, comp);
234  sort<T, StrictWeakOrdering>(++pivot, last, comp);
235 }

260 template<class T>
261 T gcd(T a, T b) {
262  // Note: We check for <= instead of < to avoid spurious compiler
263  // warnings if T is an unsigned type, i.e. warnings like "comparison
264  // of unsigned expression < 0 is always false".
265  if (a <= 0)
266  a = -a;
267  if (b <= 0)
268  b = -b;
269 
270  while (a > 0) {
271  T tmp = a;
272  a = b % a;
273  b = tmp;
274  }
275 
276  return b;
277 }
