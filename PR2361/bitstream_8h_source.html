55 template<class STREAM, int valueBits, bool isLE, bool MSB2LSB>
56 class BitStreamImpl {
57 private:
58  STREAM *_stream;
59  DisposeAfterUse::Flag _disposeAfterUse;
60 
61  uint64 _bitContainer;
62  uint8 _bitsLeft;
63  uint32 _size;
64  uint32 _pos;
65 
67  inline uint32 readData() {
68  if (isLE) {
69  if (valueBits == 8)
70  return _stream->readByte();
71  if (valueBits == 16)
72  return _stream->readUint16LE();
73  if (valueBits == 32)
74  return _stream->readUint32LE();
75  } else {
76  if (valueBits == 8)
77  return _stream->readByte();
78  if (valueBits == 16)
79  return _stream->readUint16BE();
80  if (valueBits == 32)
81  return _stream->readUint32BE();
82  }
83 
84  assert(false);
85  return 0;
86  }
87 
89  inline void fillContainer(size_t min) {
90  while (_bitsLeft < min) {
91 
92  uint64 data;
93  if (_pos + _bitsLeft + valueBits <= _size) {
94  data = readData();
95  } else {
96  // Peeking data out of bounds is well defined and returns 0 bits.
97  // This is for convenience when using speed-up techniques reading
98  // more bits than actually available. Users should call eos() to
99  // check if data was actually read out of bounds. Peeking out of
100  // bounds does not set the eos flag.
101  data = 0;
102  }
103 
104  // Move the data value to the right position in the bit container
105  if (MSB2LSB)
106  _bitContainer |= data << (64 - valueBits - _bitsLeft);
107  else
108  _bitContainer |= data << _bitsLeft;
109 
110  _bitsLeft += valueBits;
111  }
112 }
113 
115  inline static uint32 getNBits(uint64 value, size_t n) {
116  if (n == 0)
117  return 0;
118 
119  const size_t toShift = 64 - n;
120 
121  if (MSB2LSB)
122  return value >> toShift;
123  else
124  return (value << toShift) >> toShift;
125  }
126 
128  inline void skipBits(size_t n) {
129  assert(n <= _bitsLeft);
130 
131  // Shift to the next bit
132  if (MSB2LSB)
133  _bitContainer <<= n;
134  else
135  _bitContainer >>= n;
136 
137  _bitsLeft -= n;
138  _pos += n;
139  }
140 
141 public:
143  BitStreamImpl(STREAM *stream, DisposeAfterUse::Flag disposeAfterUse = DisposeAfterUse::NO) :
144  _stream(stream), _disposeAfterUse(disposeAfterUse), _bitContainer(0), _bitsLeft(0), _pos(0) {
145 
146  if ((valueBits != 8) && (valueBits != 16) && (valueBits != 32))
147  error("BitStreamImpl: Invalid memory layout %d, %d, %d", valueBits, isLE, MSB2LSB);
148 
149  _size = (_stream->size() & ~((uint32) ((valueBits >> 3) - 1))) * 8;
150  }
151 
153  BitStreamImpl(STREAM &stream) :
154  _stream(&stream), _disposeAfterUse(DisposeAfterUse::NO), _bitContainer(0), _bitsLeft(0), _pos(0) {
155 
156  if ((valueBits != 8) && (valueBits != 16) && (valueBits != 32))
157  error("BitStreamImpl: Invalid memory layout %d, %d, %d", valueBits, isLE, MSB2LSB);
158 
159  _size = (_stream->size() & ~((uint32) ((valueBits >> 3) - 1))) * 8;
160  }
161 
162  ~BitStreamImpl() {
163  if (_disposeAfterUse == DisposeAfterUse::YES)
164  delete _stream;
165  }
166 
168  uint peekBit() {
169  fillContainer(1);
170 
171  return getNBits(_bitContainer, 1);
172  }
173 
175  uint getBit() {
176  const uint b = peekBit();
177 
178  skipBits(1);
179 
180  return b;
181  }
182 
188  uint32 peekBits(size_t n) {
189  if (n > 32)
190  error("BitStreamImpl::peekBits(): Too many bits requested to be peeked");
191 
192  fillContainer(n);
193  return getNBits(_bitContainer, n);
194  }
195 
206  uint32 getBits(size_t n) {
207  if (n > 32)
208  error("BitStreamImpl::getBits(): Too many bits requested to be read");
209 
210  const uint32 b = peekBits(n);
211 
212  skipBits(n);
213 
214  return b;
215  }
216 
228  void addBit(uint32 &x, uint32 n) {
229  if (n >= 32)
230  error("BitStreamImpl::addBit(): Too many bits requested to be read");
231 
232  if (MSB2LSB)
233  x = (x << 1) | getBit();
234  else
235  x = (x & ~(1 << n)) | (getBit() << n);
236  }
237 
239  void rewind() {
240  _stream->seek(0);
241 
242  _bitContainer = 0;
243  _bitsLeft = 0;
244  _pos = 0;
245  }
246 
248  void skip(uint32 n) {
249  while (n > 32) {
250  fillContainer(32);
251  skipBits(32);
252  n -= 32;
253  }
254 
255  fillContainer(n);
256  skipBits(n);
257  }
258 
260  void align() {
261  uint32 bitsAfterBoundary = _pos % valueBits;
262  if (bitsAfterBoundary) {
263  skip(valueBits - bitsAfterBoundary);
264  }
265  }
266 
268  uint32 pos() const {
269  return _pos;
270  }
271 
273  uint32 size() const {
274  return _size;
275  }
276 
277  bool eos() const {
278  return _stream->eos() || (_pos >= _size);
279  }
280 
281  static bool isMSB2LSB() {
282  return MSB2LSB;
283  }
284 };

297 class BitStreamMemoryStream {
298 private:
299  const byte * const _ptrOrig;
300  const byte *_ptr;
301  const uint32 _size;
302  uint32 _pos;
303  DisposeAfterUse::Flag _disposeMemory;
304  bool _eos;
305 
306 public:
307  BitStreamMemoryStream(const byte *dataPtr, uint32 dataSize, DisposeAfterUse::Flag disposeMemory = DisposeAfterUse::NO) :
308  _ptrOrig(dataPtr),
309  _ptr(dataPtr),
310  _size(dataSize),
311  _pos(0),
312  _disposeMemory(disposeMemory),
313  _eos(false) {}
314 
315  ~BitStreamMemoryStream() {
316  if (_disposeMemory)
317  free(const_cast<byte *>(_ptrOrig));
318  }
319 
320  bool eos() const {
321  return _eos;
322  }
323 
324  bool err() const {
325  return false;
326  }
327 
328  int32 pos() const {
329  return _pos;
330  }
331 
332  int32 size() const {
333  return _size;
334  }
335 
336  bool seek(uint32 offset) {
337  assert(offset <= _size);
338 
339  _eos = false;
340  _pos = offset;
341  _ptr = _ptrOrig + _pos;
342  return true;
343  }
344 
345  byte readByte() {
346  if (_pos >= _size) {
347  _eos = true;
348  return 0;
349  }
350 
351  _pos++;
352  return *_ptr++;
353  }
354 
355  uint16 readUint16LE() {
356  if (_pos + 2 > _size) {
357  _eos = true;
358  if (_pos < _size) {
359  _pos++;
360  return *_ptr++;
361  } else {
362  return 0;
363  }
364  }
365 
366  uint16 val = READ_LE_UINT16(_ptr);
367 
368  _pos += 2;
369  _ptr += 2;
370 
371  return val;
372  }
373 
374  uint16 readUint16BE() {
375  if (_pos + 2 > _size) {
376  _eos = true;
377  if (_pos < _size) {
378  _pos++;
379  return (*_ptr++) << 8;
380  } else {
381  return 0;
382  }
383  }
384 
385  uint16 val = READ_LE_UINT16(_ptr);
386 
387  _pos += 2;
388  _ptr += 2;
389 
390  return val;
391  }
392 
393  uint32 readUint32LE() {
394  if (_pos + 4 > _size) {
395  uint32 val = readByte();
396  val |= (uint32)readByte() << 8;
397  val |= (uint32)readByte() << 16;
398  val |= (uint32)readByte() << 24;
399 
400  return val;
401  }
402 
403  uint32 val = READ_LE_UINT32(_ptr);
404 
405  _pos += 4;
406  _ptr += 4;
407 
408  return val;
409  }
410 
411  uint32 readUint32BE() {
412  if (_pos + 4 > _size) {
413  uint32 val = (uint32)readByte() << 24;
414  val |= (uint32)readByte() << 16;
415  val |= (uint32)readByte() << 8;
416  val |= (uint32)readByte();
417 
418  return val;
419  }
420 
421  uint32 val = READ_BE_UINT32(_ptr);
422 
423  _pos += 4;
424  _ptr += 4;
425 
426  return val;
427  }
