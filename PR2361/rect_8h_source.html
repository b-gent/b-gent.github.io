73  uint sqrDist(const Point &p) const {
74  int diffx = ABS(p.x - x);
75  if (diffx >= 0x1000)
76  return 0xFFFFFF;
77 
78  int diffy = ABS(p.y - y);
79  if (diffy >= 0x1000)

82  return uint(diffx * diffx + diffy * diffy);

222  void clip(const Rect &r) {
223  assert(isValidRect());
224  assert(r.isValidRect());
225 
226  if (top < r.top) top = r.top;
227  else if (top > r.bottom) top = r.bottom;
228 
229  if (left < r.left) left = r.left;
230  else if (left > r.right) left = r.right;
231 
232  if (bottom > r.bottom) bottom = r.bottom;
233  else if (bottom < r.top) bottom = r.top;
234 
235  if (right > r.right) right = r.right;
236  else if (right < r.left) right = r.left;
237  }

287  static bool getBlitRect(Point &dst, Rect &rect, const Rect &clip) {
288  if (dst.x < clip.left) {
289  rect.left += clip.left - dst.x;
290  dst.x = clip.left;
291  }
292 
293  if (dst.y < clip.top) {
294  rect.top += clip.top - dst.y;
295  dst.y = clip.top;
296  }
297 
298  int right = dst.x + rect.right;
299  if (right > clip.right)
300  rect.right -= right - clip.right;
301 
302  int bottom = dst.y + rect.bottom;
303  if (bottom > clip.bottom)
304  rect.bottom -= bottom - clip.bottom;
305  return !rect.isEmpty();
306  }
