117  GLint getUniformLocation(const char *uniform) const {
118  UniformsMap::iterator kv = _uniforms->find(uniform);
119  if (kv == _uniforms->end()) {
120  GLint ret = glGetUniformLocation(*_shaderNo, uniform);
121  _uniforms->setVal(uniform, ret);
122  return ret;
123  } else {
124  return kv->_value;
125  }
126  }
